動態模擬程式:

from browser import document, html, timer
  
canvas = html.CANVAS(width=400, height=400)
brython_div = document["brython_div1"]
brython_div <= canvas
ctx = canvas.getContext("2d")
  
def draw_grid():
    ctx.clearRect(0, 0, 400, 400)
    ctx.strokeStyle = "#ccc"
    ctx.lineWidth = 1
    for i in range(0, 401, 40):
        ctx.beginPath()
        ctx.moveTo(i, 0)
        ctx.lineTo(i, 400)
        ctx.stroke()
        ctx.beginPath()
        ctx.moveTo(0, i)
        ctx.lineTo(400, i)
        ctx.stroke()
     
    # 畫出邊界
    ctx.strokeStyle = "black"
    ctx.lineWidth = 2
    ctx.strokeRect(0, 0, 400, 400)
 
def draw_robot(x, y):
    draw_grid()
    ctx.fillStyle = "blue"
    ctx.beginPath()
    ctx.arc(x * 40 + 20, y * 40 + 20, 15, 0, 6.28)
    ctx.fill()
 
x, y = 0, 0
def move():
    global x
    x = (x + 1) % 10
    draw_robot(x, y)
 
draw_grid()
timer.set_interval(move, 500)
